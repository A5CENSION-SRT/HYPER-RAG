# This is a production-ready requirements file for the Multi-Agent RAG System.
# We use compatible version specifiers (~=) to allow for safe, non-breaking updates.

# --- Web Framework & Server ---
# FastAPI is the core web framework.
fastapi~=0.111.0
# Pydantic is a dependency of FastAPI but good to specify. V2 is required.
pydantic~=2.7.0
# Pydantic-settings loads config from .env files.
pydantic-settings~=2.2.0
# Uvicorn is the high-performance ASGI server. [standard] adds recommended extras.
uvicorn[standard]~=0.29.0

# --- Database & Migrations ---

# Psycopg is the driver that allows SQLAlchemy to talk to PostgreSQL.

# Alembic is used for managing database schema migrations.
